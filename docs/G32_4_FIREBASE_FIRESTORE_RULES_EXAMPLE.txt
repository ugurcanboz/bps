// Beispielregeln nur als Entwicklungsgrundlage.
// Für Produktivbetrieb müssen Rollen, Admins und Dozenten sauber über Firebase Auth Custom Claims
// oder eine serverseitige Code-Validierung abgesichert werden.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /courses/{courseId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;

      match /learners/{learnerId} {
        allow read, write: if request.auth != null;
        match /attempts/{attemptId} { allow read, write: if request.auth != null; }
        match /progress/{moduleId} { allow read, write: if request.auth != null; }
        match /events/{eventId} { allow read, write: if request.auth != null; }
        match /coach/{docId} { allow read, write: if request.auth != null; }
      }

      match /accessCodes/{codeId} {
        allow read, write: if request.auth != null;
      }
    }
  }
}
